## 3. 云函数模板

### 3.1 标准云函数模板

```js
// cloudfunctions/模块名/index.js
const cloud = require('wx-server-sdk')
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV })
const db = cloud.database()
const _ = db.command

// 统一返回工具
const { success, fail, pageResult } = require('common/utils')

/**
 * 云函数主入口
 * @param {Object} event - 调用参数
 * @param {String} event.action - 操作类型
 * @param {Object} event.data - 业务数据
 * @param {Object} context - 云函数上下文（含用户信息）
 */
exports.main = async (event, context) => {
  const { action, data = {} } = event
  const { OPENID } = cloud.getWXContext()

  try {
    // ========== 第1步：鉴权 ==========
    const user = await getUserByOpenId(OPENID)
    if (!user) return fail(401, '用户未注册')

    // ========== 第2步：路由 ==========
    switch (action) {
      case 'list':
        return await handleList(data, user)
      case 'detail':
        return await handleDetail(data, user)
      case 'create':
        return await handleCreate(data, user)
      case 'update':
        return await handleUpdate(data, user)
      default:
        return fail(400, `未知操作: ${action}`)
    }
  } catch (err) {
    console.error(`[模块名] 错误:`, err)
    return fail(500, '服务器内部错误')
  }
}

// ========== 业务处理函数 ==========

async function handleList(data, user) {
  const { page = 1, pageSize = 20, ...filters } = data

  // 权限过滤：只查当前用户有权看的数据
  const where = buildPermissionFilter(user, filters)

  const total = await db.collection('collection_name')
    .where(where).count()
  const list = await db.collection('collection_name')
    .where(where)
    .orderBy('create_time', 'desc')
    .skip((page - 1) * pageSize)
    .limit(pageSize)
    .get()

  return pageResult(list.data, total.total, page, pageSize)
}

async function handleCreate(data, user) {
  // 1. 权限校验
  if (!canCreate(user)) return fail(403, '无操作权限')

  // 2. 数据校验
  const err = validate(data)
  if (err) return fail(400, err)

  // 3. 写入数据库
  const doc = {
    ...data,
    submitter_id: user._id,
    submitter_name: user.name,
    create_time: Date.now(),
    update_time: Date.now(),
    is_deleted: false
  }
  const result = await db.collection('collection_name').add({ data: doc })

  return success({ _id: result._id }, '创建成功')
}

// ========== 辅助函数 ==========

async function getUserByOpenId(openid) {
  const result = await db.collection('users')
    .where({ openid, is_deleted: false }).get()
  return result.data[0] || null
}

function buildPermissionFilter(user, filters) {
  const where = { is_deleted: false, ...filters }

  // 按角色限制数据范围
  switch (user.role) {
    case 'class_teacher':
      where.class_id = user.class_id    // 只看自己班
      break
    case 'psychology_teacher':
      where['tags.category'] = 'psychological'  // 只看心理类
      break
    case 'admin':
    case 'principal':
    case 'vice_principal':
      // 不做限制，可看全校
      break
    default:
      where.submitter_id = user._id    // 其他人只看自己的
  }
  return where
}

function canCreate(user) {
  const allowRoles = ['admin', 'director', 'class_teacher', 'instructor']
  return allowRoles.includes(user.role)
}

function validate(data) {
  // 必填字段检查
  // 返回 null 表示通过，返回字符串表示错误信息
  return null
}

exports.main = main  // 注意：替换上面重复的 exports.main
```

> **注意**：上面模板中 `exports.main` 出现了两次（入口处和末尾），实际使用时只保留末尾的 `module.exports = { main }` 形式，或者把 `exports.main = async ...` 放在文件末尾。

### 3.2 云函数调用封装（前端）

```js
// miniprogram/utils/request.js

/**
 * 调用云函数
 * @param {String} name - 云函数名称
 * @param {Object} data - 参数 { action, data }
 * @param {Object} options - 选项 { showLoading, showError }
 * @returns {Promise<Object>} { code, message, data }
 */
async function callCloudFunction(name, data = {}, options = {}) {
  const { showLoading = true, showError = true } = options

  if (showLoading) wx.showLoading({ title: '加载中...', mask: true })

  try {
    const result = await wx.cloud.callFunction({ name, data })
    const res = result.result

    if (res.code !== 0) {
      if (showError) wx.showToast({ title: res.message, icon: 'none' })
      return res
    }
    return res
  } catch (err) {
    console.error(`[云函数调用失败] ${name}:`, err)
    if (showError) wx.showToast({ title: '网络异常，请重试', icon: 'none' })
    return { code: -1, message: '网络异常' }
  } finally {
    if (showLoading) wx.hideLoading()
  }
}

module.exports = { callCloudFunction }
```

### 3.3 前端页面调用示例

```js
// pages/student/list/index.js
const { callCloudFunction } = require('../../../utils/request')

Page({
  data: {
    studentList: [],
    total: 0,
    page: 1,
    pageSize: 20,
    isLoading: false,
    hasMore: true
  },

  onLoad() {
    this.loadStudents()
  },

  async loadStudents() {
    if (this.data.isLoading || !this.data.hasMore) return
    this.setData({ isLoading: true })

    const res = await callCloudFunction('student', {
      action: 'list',
      data: {
        page: this.data.page,
        pageSize: this.data.pageSize,
        classId: getApp().globalData.user.classId
      }
    })

    if (res.code === 0) {
      const { list, total, page, pageSize } = res.data
      this.setData({
        studentList: this.data.studentList.concat(list),
        total,
        page: page + 1,
        hasMore: page * pageSize < total,
        isLoading: false
      })
    } else {
      this.setData({ isLoading: false })
    }
  },

  onReachBottom() {
    this.loadStudents()
  }
})
```

### 5.4 数据冗余策略

| 原则 | 说明 |
|------|------|
| 宁可冗余，不轻易 lookup | 学生文档里存 `class_name`，不每次联查 |
| 冗余字段需同步更新 | 班级改名时，云函数批量更新学生文档 |
| 标签用数组存储 | `tags: ['单亲家庭', '疑似辍学']`，便于查询 |
| 脱敏字段冗余 | `id_card`（加密）+ `id_card_last4`（明文）并存 |

### 5.5 查询性能规范

| # | 规则 | 说明 |
|:---:|------|------|
| 1 | **列表查询必须带 semester** | 所有与学期关联的集合（违纪/家访/劝学/巡逻/安全/德育/违禁品/谈心/操作日志），`where` 条件首字段必须是 `semester`，确保命中复合索引 |
| 2 | **student 类查询必须带 grade_id 或 class_id** | 学生列表/标签/违纪等查询，必须带 `grade_id` 或 `class_id` 缩小扫描范围 |
| 3 | **必须分页** | 所有列表接口一律使用 `.skip()` + `.limit()`，禁止不带分页参数的全量查询 |
| 4 | **分页走索引** | 分页查询的排序字段 `orderBy` 必须在索引覆盖范围内 |
| 5 | **用 `.count()` + `total` 判断分页** | 禁止全量查回来再判断是否还有下一页，使用 `total / (page * pageSize)` 计算 `hasMore` |

```js
// ✅ 正确：带 semester + 分页，命中复合索引
async function handleList(data, user) {
  const { page = 1, pageSize = 20, semester } = data
  const where = {
    semester,                                  // ← 必须有
    ...buildPermissionFilter(user)
  }
  const total = await db.collection('discipline_records').where(where).count()
  const list = await db.collection('discipline_records')
    .where(where)
    .orderBy('create_time', 'desc')
    .skip((page - 1) * pageSize)
    .limit(pageSize)
    .get()
  return pageResult(list.data, total.total, page, pageSize)
}
```

## 7. 权限校验规范

### 7.1 核心原则

```
前端只负责隐藏按钮和页面入口
后端（云函数）负责真正的权限校验
永远不要信任前端传来的 user.role
```

### 7.2 云函数鉴权模式

```js
// 每个云函数入口必须执行
async function authenticate() {
  const { OPENID } = cloud.getWXContext()
  const user = await db.collection('users')
    .where({ openid: OPENID, is_deleted: false, status: 'approved' })
    .get()
  if (!user.data.length) throw { code: 401, message: '用户未注册或未通过审核' }
  return user.data[0]
}

// 角色检查
function requireRole(user, allowedRoles) {
  if (!allowedRoles.includes(user.role)) {
    throw { code: 403, message: '您没有操作权限' }
  }
}

// 数据范围检查（班主任只能看自己班）
function requireClassAccess(user, targetClassId) {
  if (user.role === 'class_teacher' && user.class_id !== targetClassId) {
    throw { code: 403, message: '您只能操作本班数据' }
  }
}
```

## 8. 状态机规范

### 8.1 范式

```js
// 定义状态转移表
const STATE_TRANSITIONS = {
  'pending': {
    allowedActions: ['approve', 'reject'],
    nextState: { approve: 'approved', reject: 'rejected' }
  },
  'approved': {
    allowedActions: ['store', 'revoke'],
    nextState: { store: 'stored', revoke: 'revoked' }
  },
  'stored': {
    allowedActions: ['return'],
    nextState: { return: 'returned' }
  },
  'returned': {
    allowedActions: []           // 终态，不可再操作
  },
  'rejected': {
    allowedActions: ['resubmit'],
    nextState: { resubmit: 'pending' }
  }
}

// 检查是否允许操作
function canTransition(currentStatus, action) {
  const state = STATE_TRANSITIONS[currentStatus]
  if (!state) return false
  return state.allowedActions.includes(action)
}

// 执行状态转移
function transition(currentStatus, action) {
  if (!canTransition(currentStatus, action)) {
    throw { code: 422, message: `状态 ${currentStatus} 不允许执行 ${action} 操作` }
  }
  return STATE_TRANSITIONS[currentStatus].nextState[action]
}
```

### 8.2 禁止的写法

```js
// ❌ 禁止：零散的 if 判断状态
if (record.status === 'pending') {
  // 审核逻辑
} else if (record.status === 'approved') {
  // ...
}
// 散落各处，容易遗漏状态、不同步

// ✅ 正确：统一的状态机驱动
const newStatus = transition(record.status, 'approve')
await db.collection('collection').doc(record._id).update({
  data: { status: newStatus, update_time: Date.now() }
})
```

## 11. 禁止事项清单

| # | 禁止事项 | 原因 | 正确做法 |
|:---:|------|------|------|
| 1 | 前端做鉴权 | 可被绕过 | 云函数中校验角色+数据范围 |
| 2 | 源码中硬编码密钥 | 安全漏洞 | 用云函数环境变量 |
| 3 | 明文存身份证 | 数据泄露风险 | AES 加密，冗余 `_last4` 脱敏字段 |
| 4 | 全量加载列表 | 性能崩溃 | `.skip()` + `.limit()` 分页 |
| 5 | 循环中调云函数 | 性能极差 | 批量接口或 `Promise.all` |
| 6 | 用 lookup 做联表 | 性能差 | 冗余字段 |
| 7 | 零散 if 判断状态 | 遗漏分支 | 状态机模式 |
| 8 | 前端生成 Excel/PDF | 性能差 | 云函数生成 |
| 9 | 自定义复杂 CSS | 浪费时间 | 用 WeUI |
| 10 | 用 `wx.getUserInfo` 获取用户信息 | 已废弃 | 用 `<button open-type="getPhoneNumber">` |
| 11 | 一行超长代码不换行 | 可读性差 | 120 字符换行 |
| 12 | 不处理异步错误 | 静默失败 | `try/catch` 包裹所有 `await` |
| 13 | 不带 `semester` 条件查询历史数据集合 | 全表扫描，索引失效 | `where` 首字段必须为 `semester` |
